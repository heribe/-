\#归并排序 

def mergeSort(l): 

if len(l)==1: 

return l 

mid = int(len(l)/2) 

l1,l2 = l[:mid],l[mid:] 

l1 = mergeSort(l1) 

l2 = mergeSort(l2) 

i,j = 0,0 

l = [] 

while i<len(l1) and j<len(l2): 

if l1[i]<l2[j]: 

l.append(l1[i]) 

i+=1 

else: 

l.append(l2[j]) 

j+=1 

if i==len(l1): 

l.extend(l2[j:]) 

else: 

l.extend(l1[i:]) 

return l 

\# test

l=mergeSort(l) 

print(l) 

\# 时间复杂度=O(nlogn) 

\# 空间复杂度=O(n)

def quickSort(l,left,right): 

if left>=right: 

return 

mid = random.randint(left, right) 

l[left],l[mid]=l[mid],l[left] 

i,j = left,right 

while i<j: 

while i<j and l[i]<l[j]: 

j-=1 

if i==j: 

break 

l[i],l[j]=l[j],l[i] 

i+=1 

while i<j and l[i]<l[j]: 

i+=1 

if i==j: 

break 

l[i],l[j]=l[j],l[i] 

j-=1 

quickSort(l,left,i-1) 

quickSort(l,i+1,right)

def insertSort(l): 

if len(l)==1: 

return 

for i in range(1,len(l)): 

j = i-1 

while j>=0: 

if l[j]<l[i]: 

break 

l[j],l[i]=l[i],l[j] 

i=j 

j-=1

def bubbleSort(l): 

if len(l)==1: 

return 

for j in range(len(l)-1,0,-1): 

for i in range(j): 

if l[i]>l[j]: 

l[i],l[j] = l[j],l[i]

ef solution(l,k): 

mid = random.randint(0,len(l)-1) 

l[0],l[mid]=l[mid],l[0] 

i,j = 0,len(l)-1 

while i<j: 

while i<j and l[i]<l[j]: 

j-=1 

if i==j: 

break 

l[i],l[j]=l[j],l[i] 

i+=1 

while i<j and l[i]<l[j]: 

i+=1 

if i==j: 

break 

l[i],l[j]=l[j],l[i] 

j-=1 

if i+1==k:
