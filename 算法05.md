   ```python
   # python3
   class Node():
       def __init__(self,key=None):
           self.key = key
           self.left = None
           self.right = None
   
   class binarySearchTree():
       def __init__(self):
           self.head = None
   
       def insert(self,key):
           if not self.head:
               self.head = Node(key)
           else:
               t = self.head
               self.insertNode(t,Node(key))
   
       def insertNode(self,t,node):
           while t:
               if node.key>t.key:
                   if not t.right:
                       t.right,t = node,None
                   else:
                       t = t.right
               else:
                   if not t.left:
                       t.left,t = node,None
                   else:
                       t = t.left
   
       def delete(self,key):
           t = self.find(key)
           if not t:
               print('(delete)%s is not in this tree'%key)
               return False
           pre,dir = t[0],t[1]
           if dir=='right':
               left = pre.right.left
               pre.right = pre.right.right
               if left:
                   self.insertNode(pre,left)
           elif dir=='left':
               right = pre.left.right
               pre.left = pre.left.left
               if right:
                   self.insertNode(pre,right)
           else:
               print('dir error')
               return False
   
       # 如果找到了，返回一个有两个元素的列表，两个元素为找到节点的父节点和此节点是父节点的左孩子还是右孩子，实例：[pre,'right']
       # 如果没找到，则返回False
       def find(self,key):
           if key == self.head.key:
               pre = Node(None)
               pre.right = self.head
               return [pre,'right']
           pre = self.head
           while pre:
               if key>pre.key:
                   if pre.right and pre.right.key==key:
                       return [pre,'right']
               else:
                   if pre.left and pre.left.key==key:
                       return [pre,'left']
               pre = pre.right
           print('(find)%s not in this tree'%key)
           return False

   bst.insert(19)
   bst.insert(18)
   bst.delete(19)
   bst.delete(19)

   print(bst.find(20))
   ```

3. 实现二叉树前、中、后序以及按层遍历

   ```python
   # python3
   # 实现二叉树前、中、后序以及按层遍历
   import queue
   class Note():
       def __init__(self,key=None):
           self.key = key
           self.right = None
           self.left = None
   
   class BinaryTree():
       def __init__(self,head = Note(None)):
           self.head = head
   
       # 前序遍历
       # f为对key进行操作的函数，例子：
       # def f(x):
       #     print(x)
       def preOrderTraversal(self,f):
           node = self.head
           self.preOT(node,f)
   
       def preOT(self,node,f):
           if node:
               f(node.key)
               self.preOT(node.left,f)
               self.preOT(node.right,f)
   
       # 中序遍历
       def inOrderTraversal(self,f):
           node = self.head
           self.inOT(node,f)
   
       def inOT(self,node,f):
           if node:
               self.inOT(node.left,f)
               f(node.key)
               self.inOT(node.right,f)
   
       # 后序遍历
       def postOrderTraversal(self,f):
           node = self.head
           self.postOT(node,f)
   
       def postOT(self,node,f):
           if node:
               self.postOT(node.left,f)
               self.postOT(node.right,f)
               f(node.key)
       
       # 按层遍历：
       def layerTraver(self,f):
           qu = queue.Queue()
           qu.put(self.head)
           while not qu.empty():
               note = qu.get()
               f(note.key)
               if note.left:
                   qu.put(note.left)
               if note.right:
                   qu.put(note.right)
           
   
   
   #test
   tree = Note(5)
   tree.right = Note(3)
   tree.left = Note(4)
   tree.right.right = Note(9)
   tree.right.left = Note(1)
   tree.left.left = Note(6)
   tree.left.right = Note(2)
   tree.left.right.right = Note(8)
   #    5
   #  4   3
   # 6 2 1 9
   #   8
   def f(x):
       print(x)
   bt = BinaryTree(tree)
   print('前序遍历')
   bt.preOrderTraversal(f)
   print('中序遍历')
   bt.inOrderTraversal(f)
   print('后序遍历')
   bt.postOrderTraversal(f)
   print('按层遍历')
   bt.layerTraver(f)
   ```

```python
import random
# 寻找最大的k个值
def TopK(k,l):
    c = l[:k]
    heap = MinHeap()
    heap.listToMinHeap(c)
    for i in range(k,len(l)):
        if l[i]>heap.heap[0]:
            heap.get()
            heap.put(l[i])
    return heap.heap
```
